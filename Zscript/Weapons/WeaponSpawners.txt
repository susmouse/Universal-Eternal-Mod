Class PistolSpawner : EventHandler {
    override void CheckReplacement (replaceEvent e) {
        if (e.Replacee.GetClassName() == "Pistol") {
            let playerCls = players[consoleplayer].cls;
            if (playerCls == "HellCrusher")
                e.Replacement = "Executioner";
				
            else if (playerCls == "HellCrusherIP")
                e.Replacement = "InfinityMagnum";
				
			else if (playerCls == "HellCrusherSSG")
                e.Replacement = "BoomStick";
				
			else if (playerCls == "HellCrusherDL")
                e.Replacement = "BlackTalons";
				
			else if (playerCls == "HellCrusherRV")
                e.Replacement = "MammothTusk";
        }
    }
}
